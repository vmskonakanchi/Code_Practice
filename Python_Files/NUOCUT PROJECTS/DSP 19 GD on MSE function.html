def grad(x, y, thetas):
    
    n = y.size
    
    theta0_slope = (-2/n)* sum(y - thetas[0] - thetas[1]*x)
    
    theta1_slope = (-2/n)* sum((y - thetas[0] - thetas[1]*x)*x)
    
    
    return np.array([theta0_slope[0], theta1_slope[0]])
